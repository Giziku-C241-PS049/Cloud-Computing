Giziku API Documentation

GitHub Repository

Base URL

http://127.0.0.1:8000

Endpoints

Register User

POST /register

Request Body:

{
  "email": "string",
  "password": "string",
  "username": "string"
}

Response:

{
  "status": "success",
  "message": "User registered successfully",
  "data": {
    "email": "string",
    "username": "string"
  }
}

Error Response:

{
  "status": "error",
  "message": "User already exists" // atau
}
{
    "status": "error",
    "message": "Server error",
    "error": "error message"
  }

Login User

POST /login

Request Body:

{
  "email": "string",
  "password": "string"
}

Response:

{
  "status": "success",
  "message": "User logged in successfully",
  "data": {
    "email": "string",
    "username": "string",
    "token": "string"
  }
}

Error Response:

{
  "status": "error",
  "message": "Invalid credentials"
}
{
    "status": "error",
    "message": "Server error",
    "error": "error message"
  }

Update Username

PUT /update-username

Headers:

Authorization: Bearer <token>

Request Body:

{
  "newUsername": "string"
}

Response:

{
  "status": "success",
  "message": "Username updated successfully",
  "data": {
    "email": "string",
    "newUsername": "string"
  }
}

Error Response:

{
  "status": "error",
  "message": "New username is required" 
}
{
    "status": "error",
    "message": "Server error",
    "error": "error message"
  }
